genre_cat = []
genres = ["Pop", "Rap", "Hip Hop", "Country", "Emo", "Rock", "Metal", "Indie", "EDM/Electro", "R&B", "Soul", "Other"]
count = [0] * 12

for i in song_data['top genre']:
    if "rap" in i:
        genre_cat.append("rap")
        count[1] += 1
    elif "hip hop" in i:
        genre_cat.append("hip hop")
        count[2] += 1
    elif "country" in i:
        genre_cat.append("country")
        count[3] += 1
    elif "emo" in i:
        genre_cat.append("emo")
        count[4] += 1
    elif "rock" in i:
        genre_cat.append("rock")
        count[5] += 1
    elif "metal" in i:
        genre_cat.append("metal")
        count[6] += 1
    elif "indie" in i:
        genre_cat.append("indie")
        count[7] += 1
    elif "r&b" in i:
        genre_cat.append("r&b")
        count[9] += 1
    elif "soul" in i:
        genre_cat.append("soul")
        count[10] += 1
    elif ("edm" or "elect") in i:
        genre_cat.append("edm")
        count[8] += 1
    elif "pop" in i:
        genre_cat.append("pop")
        count[0] += 1
    else:
        genre_cat.append("other")
        count[11] += 1
        
song_data["genre category"] = genre_cat
